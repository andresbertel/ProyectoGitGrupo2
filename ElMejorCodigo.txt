void Atender(int& frente, int& final, int* cola);    //eliminar un elemento de la cola
void Listar(int frente, int final, int* cola);
const int MaxCola = 10;  // tama?o m?ximo de la cola

int final = -1;    // Guarda la posicion del ultimo elemento de la cola

void Insertar(int& frente, int& final, int* cola)   // Insercion de un elemento de la cola
{
	srand(time(NULL));
	if ((frente == 0 && final + 1 == MaxCola) || (final + 1 == frente)) {
		std::cout << "La cola esta llena!!\a\n";
	}
	else
	{
		dato = rand() % 50;
		std::cout << "\n\nElemento a insertar: " << dato << "\n";
		if (frente == -1) { //la cola estaba vacia

			frente = final = 0;
		}
		else { // caso general
			final++;
			if (final == MaxCola) // se da "la vuelta al inicio"
				final = 0;
		}
		cola[final] = dato;
	}
	std::cin.get();
}
void Atender(int& frente, int& final, int* cola) {   //eliminar un elemento de la cola
	if (frente == -1)
		std::cout << "La cola esta vacia!!\a";
	else {
		std::cout << "\nAtendido: " << cola[frente] << "\n";
		if (frente == final)   // hay solo un elemento
			frente = final = -1;   // cola queda vacia
		else {
			frente++;
			if (frente == MaxCola)
				frente = 0;
		}
	}
	std::cin.get();
}
void Listar(int frente, int final, int* cola) {  // Mostrar los elementos de la cola
	int i;
	if (frente == -1)
		std::cout << "\nLa Cola esta vacia \a\n";
	else {
		std::cout << "Los elementos de la cola son:";
		if (frente <= final)  // cola lineal
			for (i = frente; i <= final; i++)
				std::cout << cola[i] << "\t";
		else {       // cola circular
			for (i = frente; i < MaxCola; i++)
				std::cout << cola[i] << "\t";
			for (i = 0; i <= final; i++)
				std::cout << cola[i] << "\t";
		}
	}
	
}
